#include<iostream>h
#include<string>
using namespace std;
void paixu(int *a, int n)
{
	int i, j, t;
	if (n < 2)return;
	int p = a[n / 2];
	for (i = 0,j=n-1;i<j;i++,i--)
		{
			while (a[i]<p)
				i++;
			while (a[j]>p)
				j--;
			t = a[i]; a[i] = a[j]; a[j] = t;
		}
	paixu(a, i);
	paixu(a + i, n-i);
}
int main()
{
	int a[10];
	int i,m;
	cout << "请输入数字个数";
	cin >> m;
	for (i = 0; i<m; i++)
	{
		cin >> a[i];
	}
	paixu(a, m);
	for (i = 0; i < m; i++)
		cout << a[i];
	system("pause");
	return 0;
}
