#include<iostream>
using namespace std;
void quick(int, int);
int a[100];
int main()
{
	int n, i;
	cin >> n;
	for (i = 1; i <= n; i++)
		cin >> a[i];
	quick(0, n);
	for (i = 1; i <= n; i++)
		cout << a[i] <<"";
	cout << endl;
}
void quick(int l, int s)
{
	int i, j, mid, k;
	i = l;
	j = s;
	mid = a[(l + s) / 2];
	do
	{
		while (a[i] < mid)i++;
		while (a[j] > mid)j--;
		if (i <= j)
		{
			k = a[i];
			a[i] = a[j];
			a[j] = k;
			i++;
			j--;
		}
	} while (i <= j);
	if (l < j)quick(l, j);
	if (i < s)quick(i, s);
}
