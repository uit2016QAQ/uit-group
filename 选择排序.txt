
#include<stdio.h>

void selete(int a[],int n)
{
	int i,j,k,temp;
	for(i=0;i<=n;i++)
	{
     	k=i;
		for(j=i+1;j<=n;j++)
		{
			if(a[j]<a[k])
				k=j;
			if(k!=i)
			{
				temp=a[i];
				a[i]=a[k];
				a[k]=temp;
			}
		}
	}
       for(i=1;i<=n;i++)
            {

            }
}
